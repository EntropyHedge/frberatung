[Unit]
Description=Celery Worker Service
After=network.target redis.service postgresql.service
Requires=redis.service

[Service]
Type=simple
User=zielzone
Group=zielzone
EnvironmentFile=/etc/conf.d/celery
EnvironmentFile=/home/zielzone/zielzone/.env
WorkingDirectory=/home/zielzone/zielzone
Environment=PYTHONPATH=/home/zielzone/zielzone
ExecStart=ExecStart=/home/zielzone/zielzone/venv/bin/celery -A app.make_celery worker --concurrency=4 --max-tasks-per-child=100 --loglevel=INFO
Restart=always
RestartSec=10

# Logging
StandardOutput=append:/var/log/celery/worker.log
StandardError=append:/var/log/celery/worker.error.log

[Install]
WantedBy=multi-user.target
